        /**

         * Helper function to dynamically load an external chat script (Utterances).

         * @param {string} containerId - ID of the element to inject the script into.

         * @param {Object} attrs - Attributes for the script tag.

         */

        function loadExternalChat(containerId, attrs) {

            const container = document.getElementById(containerId);

            if (!container) return;

            

            // 1. Clear previous content (important for dynamic view switching)

            container.innerHTML = '';


            // 2. Create the script element

            const script = document.createElement("script");

            for (const key in attrs) {

                script.setAttribute(key, attrs[key]);

            }

            script.setAttribute('crossorigin', 'anonymous');

            script.setAttribute('async', 'true');


            // 3. Append the script to the target container

            container.appendChild(script);

        }

        function renderView(viewName) {

                link.classList.add('text-primer-fg/70');

            });

            const activeLink = document.getElementById(`nav-${viewName}`);

            if (activeLink) {

                activeLink.classList.add('font-bold', 'text-primer-hover');

            if (viewName === VIEW_ROOMS) {

        // --- Firebase & Auth Initialization ---


        function initFirebase() {

            

            try {

                app = initializeApp(firebaseConfig);

                auth = getAuth(app);

                

                // 1. Attempt Sign-in

                if (initialAuthToken) {

                    signInWithCustomToken(auth, initialAuthToken)

                        .catch(error => {

                            console.error("Custom token sign-in failed. Falling back to anonymous.", error);

                            signInAnonymously(auth);

                        });

                } else {

                    signInAnonymously(auth);

                }


                // 2. Set up Auth State Listener

                onAuthStateChanged(auth, (user) => {

                    isAuthReady = true;

                    let statusText;

                    const isMock = firebaseConfig.apiKey === 'mock-key';


                    if (user) {

                        userId = user.uid;

                        statusText = `User: ${userId.substring(0, 8)}... (Authenticated)`;

                    } else {

                        userId = crypto.randomUUID(); // Fallback ID

                        statusText = 'User: Anonymous';

                    }

                    

                    if (isMock) {

                        statusText = 'User: Mock Auth (Live Chat Only)';

                    }


                    authStatusElement.textContent = statusText;

                    

                    // Render the default view (Rooms)

                    renderView(currentView);

                });


            } catch (e) {

                console.error("Firebase initialization failed:", e);

                authStatusElement.textContent = 'ERROR: Init Failed';

                isAuthReady = true;

                renderView(VIEW_ROOMS);

            }

        }
